HOW TO CUSTOMIZE YOUR LINUX TERMINAL TO DISPLAY CUSTOM PROMPT

# Run the following command

cat >> ~/.bashrc << 'EOF'
parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/(\1)/'
}
export PS1='\[\033[1;36m\]\W\[\033[1;32m\]$(parse_git_branch)\[\033[00m\]$ '
EOF

source ~/.bashrc

# Or close your terminal and open it again.
